public int sol(int n, int k, int[] arr) {

        int answer = -1;
        TreeSet<Integer> tree = new TreeSet<>(Collections.reverseOrder());

        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                for (int l = j + 1; l < n; l++) {
                    tree.add(arr[i] + arr[j] + arr[l]);
                }
            }

        }
        int cnt = 0;
        for (int x : tree) {
            cnt++;
            if (cnt == k) return x;
        }
        return answer;
    }
