    public String solution(String need, String plan) {
        String answer = "YES";
        Queue<Character> q = new LinkedList<>();

        for (char x : need.toCharArray()) q.offer(x);
        for(char x : plan.toCharArray()) {
            if(q.contains(x)) {
                if(x != q.poll()) return "NO";
            }
            if(!q.isEmpty()) return "NO";
        }
        return answer;
    }
