public int sol(String str) {
        int answer = 0;

        Stack<Integer> stack = new Stack<>();
        for (char x : str.toCharArray()) {
            if (Character.isDigit(x)) stack.push(x - '0');
            else {
                int a = stack.pop();
                int b = stack.pop();
                switch (x) {
                    case '+':
                        stack.push(b + a);
                        break;
                    case '-':
                        stack.push(b - a);
                        break;
                    case '*':
                        stack.push(b * a);
                        break;
                    case '/':
                        stack.push(b / a);
                        break;
                }
            }
        }
        answer = stack.get(0);
        return answer;
    }
