    public int sol(int a, int b) {
        int answer = 0;
        Queue<Integer> q = new LinkedList<>();
        
        for (int i = 1; i <= a; i++) q.offer(i);
        while(!q.isEmpty()) {
            for (int j = 0; j < b; j++) q.offer(q.poll());
            q.poll();
            if(q.size() == 1) answer=q.poll();
        }
        return answer;
    }
