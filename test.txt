    public String sol(String a) {
        String answer = "YES";
        Stack<Character> stack = new Stack<>();
        for(char x : a.toCharArray()) {
            if(x == '(') stack.push(x);
            else {
                if(stack.isEmpty()) return "NO";
                stack.pop();
            }
        }
        if(!stack.empty()) return "NO";
        return answer;
    }
